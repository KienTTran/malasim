118 template <typename T>
119 T random_uniform(T from, T to) {
120 static_assert(std::is_arithmetic<T>::value,
121 "Template parameter T must be numeric.");
122
123 if (!rng_) {
124 throw std::runtime_error("Random number generator not initialized.");
125 }
126
127 if (from >= to) {
128 throw std::invalid_argument("Parameter 'from' must be less than 'to'.");
129 }
130
131 if constexpr (std::is_integral<T>::value) {
132 // gsl_rng_uniform_int takes an unsigned long as upper bound
133 // Ensure that (to - from) does not exceed the maximum value of unsigned
134 // long
135 unsigned long range = static_cast<unsigned long>(to - from);
136 // Note: gsl_rng_uniform_int handles range values up to the maximum of
137 // unsigned long
138 unsigned long value = gsl_rng_uniform_int(rng_.get(), range);
139 return static_cast<T>(from + value);
140 } else {
141 double value = gsl_ran_flat(rng_.get(), static_cast<double>(from),
142 static_cast<double>(to));
143 return static_cast<T>(value);
144 }

229 template <typename T>
230 T random_normal_truncated(T mean, double standard_deviation,
231 double truncation_limit = 3.0, // NOLINT
232 int max_attempts = 1000) {
233 static_assert(std::is_arithmetic<T>::value,
234 "Template parameter T must be numeric.");
235 if (!rng_) {
236 throw std::runtime_error("Random number generator not initialized.");
237 }
238 double value = gsl_ran_gaussian(rng_.get(), standard_deviation);
239 int attempts = 0;
240 while (std::abs(value) > truncation_limit * standard_deviation) {
241 if (attempts++ >= max_attempts) {
242 throw std::runtime_error(
243 "Failed to generate a truncated normal value within the maximum "
244 "number of attempts.");
245 }
246 value = gsl_ran_gaussian(rng_.get(), standard_deviation);
247 }
248 if constexpr (std::is_integral<T>::value) {
249 return static_cast<T>(mean + std::round(value));
250 } else {
251 return static_cast<T>(mean + value);
252 }
253 }
