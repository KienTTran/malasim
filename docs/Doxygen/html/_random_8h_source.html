119 template <typename T>
120 T random_uniform(T from, T to) {
121 static_assert(std::is_arithmetic<T>::value,
122 "Template parameter T must be numeric.");
123
124 if (!rng_) {
125 throw std::runtime_error("Random number generator not initialized.");
126 }
127
128 if (from >= to) {
129 throw std::invalid_argument("Parameter 'from' must be less than 'to'.");
130 }
131
132 if constexpr (std::is_integral<T>::value) {
133 // gsl_rng_uniform_int takes an unsigned long as upper bound
134 // Ensure that (to - from) does not exceed the maximum value of unsigned
135 // long
136 unsigned long range = static_cast<unsigned long>(to - from);
137 // Note: gsl_rng_uniform_int handles range values up to the maximum of
138 // unsigned long
139 unsigned long value = gsl_rng_uniform_int(rng_.get(), range);
140 return static_cast<T>(from + value);
141 } else {
142 double value = gsl_ran_flat(rng_.get(), static_cast<double>(from),
143 static_cast<double>(to));
144 return static_cast<T>(value);
145 }

160 template <typename T>
161 T random_normal(T mean, double standard_deviation) {
162 static_assert(std::is_arithmetic<T>::value,
163 "Template parameter T must be numeric.");
164
165 if (!rng_) {
166 throw std::runtime_error("Random number generator not initialized.");
167 }
168
169 if (standard_deviation <= 0) {
170 throw std::invalid_argument(
171 "Parameters 'standard_deviation' must be greater than 0.");
172 }
173 double value = mean + gsl_ran_gaussian(rng_.get(), standard_deviation);
174
175 if constexpr (std::is_integral<T>::value) {
176 return static_cast<T>(std::round(value));
177 } else {
178 return static_cast<T>(value);

204 template <typename T>
205 T random_normal_truncated(T mean, double standard_deviation,
206 double truncation_limit = 3.0, // NOLINT
207 int max_attempts = 1000) {
208 static_assert(std::is_arithmetic<T>::value,
209 "Template parameter T must be numeric.");
210 if (!rng_) {
211 throw std::runtime_error("Random number generator not initialized.");
212 }
213 double value = gsl_ran_gaussian(rng_.get(), standard_deviation);
214 int attempts = 0;
215 while (std::abs(value) > truncation_limit * standard_deviation) {
216 if (attempts++ >= max_attempts) {
217 throw std::runtime_error(
218 "Failed to generate a truncated normal value within the maximum "
219 "number of attempts.");
220 }
221 value = gsl_ran_gaussian(rng_.get(), standard_deviation);
222 }
223 if constexpr (std::is_integral<T>::value) {
224 return static_cast<T>(mean + std::round(value));
225 } else {
226 return static_cast<T>(mean + value);
227 }
